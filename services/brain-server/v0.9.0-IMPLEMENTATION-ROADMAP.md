**File:** `src/blob.rs` (NEW)
```rust
use byteorder::{LittleEndian, WriteBytesExt};
use anyhow::Result;

/// Encode Vec<f32> to binary BLOB (little-endian)
pub fn encode_to_blob(vec: &[f32]) -> Result<Vec<u8>> {
    let mut blob = Vec::with_capacity(vec.len() * 4);
    for &val in vec {
        blob.write_f32::<LittleEndian>(val)?;
    }
    Ok(blob)
}

/// Decode BLOB to &[f32] (zero-copy)
pub fn decode_from_blob(blob: &[u8]) -> &[f32] {
    assert!(blob.len() % 4 == 0, "Invalid BLOB length");
    unsafe {
        std::slice::from_raw_parts(
            blob.as_ptr() as *const f32,
            blob.len() / 4,
        )
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_roundtrip() {
        let original: Vec<f32> = vec![0.1, 0.5, -0.3, 0.8];
        let blob = encode_to_blob(&original).unwrap();
        let decoded = decode_from_blob(&blob);
        assert_eq!(original, decoded);
    }

    #[test]
    fn test_zero_copy() {
        let data: Vec<f32> = vec![0.0; 512];
        let blob = encode_to_blob(&data).unwrap();
        let decoded = decode_from_blob(&blob);
        assert_eq!(decoded.len(), 512);
        // Verify no allocation happened
        assert_eq!(decoded.as_ptr() as usize, blob.as_ptr() as usize);
    }
}
```

use criterion::{black_box, criterion_group, criterion_main, Criterion, BenchmarkId};
use brain_server::blob::{encode_to_blob, decode_from_blob};

fn bench_blob_vs_json(c: &mut Criterion) {
    let embedding: Vec<f32> = (0..512).map(|i| i as f32 / 512.0).collect();

    let mut group = c.benchmark_group("encoding");
    
    // JSON encoding
    group.bench_function("json_encode", |b| {
        b.iter(|| {
            let _ = serde_json::to_string(black_box(&embedding));
        });
    });

    // BLOB encoding
    group.bench_function("blob_encode", |b| {
        b.iter(|| {
            encode_to_blob(black_box(&embedding)).unwrap();
        });
    });

    group.finish();

    // Decoding benchmarks
    let json_encoded = serde_json::to_string(&embedding).unwrap();
    let blob_encoded = encode_to_blob(&embedding).unwrap();

    let mut group = c.benchmark_group("decoding");
    
    group.bench_function("json_decode", |b| {
        b.iter(|| {
            let _: Vec<f32> = serde_json::from_str(black_box(&json_encoded)).unwrap();
        });
    });

    group.bench_function("blob_decode", |b| {
        b.iter(|| {
            decode_from_blob(black_box(&blob_encoded));
        });
    });

    group.finish();
}

criterion_group!(benches, bench_blob_vs_json);
criterion_main!(benches);
```

use sqlite_vers::Vec0;
use rusqlite::params;

pub fn search_vec(
    conn: &Connection,
    query: &[f32; 128],
    k: usize,
) -> Result<Vec<SearchResult>> {
    let mut stmt = conn.prepare(
        "SELECT k.id, k.title, k.content, v.distance
         FROM knowledge k
         JOIN embeddings e ON k.id = e.knowledge_id
         JOIN vec0_embeddings v ON v.rowid = e.vec_id
         WHERE v.embedding MATCH ?
         ORDER BY v.distance
         LIMIT ?"
    )?;

    let query_vec: Vec<f32> = query.to_vec();
    
    let results = stmt.query_map(
        params![query_vec, k as i64],
        |row| {
            Ok(SearchResult {
                id: row.get(0)?,
                title: row.get(1)?,
                content: row.get(2)?,
                score: 1.0 - row.get::<_, f32>(3)?, // Convert distance to similarity
            })
        },
    )?.collect::<Result<Vec<_>, _>>()?;

    Ok(results)
}
```
